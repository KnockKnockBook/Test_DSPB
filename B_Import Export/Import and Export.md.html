Import data

Import from flat file - csv, tab-delimited, etc

# Create a filepath
import_path = "/Users/jonahbrown-joel/Library/Mobile Documents/com~apple~CloudDocs/Data Science Cookbook/Datasets/Import Examples/"
import_filepath = import_path + "titanic3.csv"

# import a csv file
DF_Titanic = pd.read_csv(import_path + "titanic3.csv")

# import a tab-delimited text file
DF_Titanic = pd.read_csv(import_path + "titanic3.txt"
                , delimiter = "\t")

# Specify specific column names to be imported
DF_Titanic = pd.read_csv(import_path + "titanic3.csv"
                ,  usecols = ['pclass', 'survived' , 'name'] ) # use 'usecols', and specify a list of columns to import

# Set the datatypes of columns in your imported dataframe  
DF_Titanic = pd.read_csv(import_path + "titanic3.csv",
                        dtype = {'pclass'   :  'int32' ,   # use 'dtype', and specify a dictionary of column names and data types for each column
                                 'survived' :  'int64' ,
                                 'name'     :  'object' }
                         )

Import from Excel

filename    = "titanic3.xls"
import_path = "/Users/jonahbrown-joel/Library/Mobile Documents/com~apple~CloudDocs/Data Science Cookbook/Datasets/Import Examples/"
import_filepath = import_path + filename


# Specify an excel filepath
DF_Titanic = pd.read_excel(import_filepath)

# Specify a sheetname - useful if your file contains multiple sheets
DF_Titanic = pd.read_excel(import_filepath
                         , sheet_name= "titanic3")

# Specify specific column names to be imported, use 'usecols' - just like read_csv()
DF_Titanic = pd.read_excel(import_filepath
                         , sheet_name = "titanic3"
                         , usecols = ['pclass', 'survived' , 'name']  # use 'usecols', and specify a list of columns to import
                         )

# Set the datatypes of columns in your imported dataframe, use 'dtype' - just like read_csv()  
DF_Titanic = pd.read_excel(import_filepath
                     , sheet_name = "titanic3"
                     , dtype = { 'survived' :  'int32' ,
                                 'name'     :  'object' }
                         )

Export data

Export to CSV, Text file, or other delimiter

##### Export to CSV file (.csv)
export_path = "/Users/jonahbrown-joel/Library/Mobile Documents/com~apple~CloudDocs/Data Science Cookbook/Datasets/Export Examples/"
DF_Titanic.to_csv(export_path + "titanic3.csv", index = False)

##### Export to Tab-Delimited text file (.txt)
DF_Titanic.to_csv(export_path + "tab_titanic.txt"
                , sep ='\t'
        , index = False)

##### Export to Carrot-Delimited text file (.txt), or other delimiter
# Changing the parameter sep = 'x' to the delimiter of your dreams
DF_Titanic.to_csv(export_path + "other_delimiter_titanic.txt"
                , sep ='^'
        , index = False)

# specify which columns you want to export
DF_Titanic.to_csv("/Users/jonahbrown-joel/Library/Mobile Documents/com~apple~CloudDocs/Data Science Cookbook/Datasets/Export Examples/test2.csv"
                  , columns = ['pclass', 'survived']  # specify which columns you want to export
                  , index = False)

Export to Excel

# use 'to_excel()' and specify a filepath and sheetname
export_path = "/Users/jonahbrown-joel/Library/Mobile Documents/com~apple~CloudDocs/Data Science Cookbook/Datasets/Export Examples/"
DF_Titanic.to_excel(export_path + "titanic3.xlsx"
                , sheet_name ="my_sheet_name")

# Save multiple DFs to multiple sheets
with pd.ExcelWriter('/Users/jonahbrown-joel/Library/Mobile Documents/com~apple~CloudDocs/Data Science Cookbook/Datasets/Export Examples/mulitple_sheets.xlsx') as writer:  
    DF_Titanic.to_excel(writer, sheet_name='Sheet_name_1')
    DF_Titanic.to_excel(writer, sheet_name='Sheet_name_2')

# Append/Add new DF to existing Excel file without overwriting it
with pd.ExcelWriter('/Users/jonahbrown-joel/Library/Mobile Documents/com~apple~CloudDocs/Data Science Cookbook/Datasets/Export Examples/existing_sheet.xlsx'
                    , mode='a' # specify the 'append mode' with 'a'
                    , engine="openpyxl") as writer:  
    DF_Titanic.to_excel(writer, sheet_name='Appended_Sheet')
